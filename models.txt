Advertiser Table:

id: INT                 AUTO_INCREMENT
name: VARCHAR (30)      NOT NULL
views: INT              DEFAULT 0
clicks: INT             DEFAULT 0
PRIMARY KEY (id)





Ad Table:

id: INT                 AUTO_INCREMENT
title: VARCHAR (50)     NOT NULL
img_url: VARCHAR (300)  NOT NULL
link: VARCHAR (300)     NOT NULL
views: INT              DEFAULT 0
clicks: INT             DEFAULT 0
PRIMARY KEY (id)
FOREIGN KEY (Advertiser_id)
